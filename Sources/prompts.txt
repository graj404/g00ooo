# PI Steering Angle Observer

Kp = 0.8      # Proportional gain — how fast to track

Ki = 0.1      # Integral gain — corrects steady state error

dt = 0.05     # 20Hz from AS5600

estimated_angle = 0.0

integral_error  = 0.0

MAX_STEERING_RATE = 120  # degrees/sec (physical limit of kart steering)

def pi_steering_observer(raw_angle, estimated_angle, integral_error, dt):

    # Step 1 — Clamp raw reading to physical steering rate limit

    max_change = MAX_STEERING_RATE * dt      # max degrees per timestep

    clamped_angle = max(min(raw_angle,

                    estimated_angle + max_change),

                    estimated_angle - max_change)

    # Step 2 — PI controller

    error          = clamped_angle - estimated_angle

    integral_error += error * dt             # I term accumulates over time

    integral_error  = max(min(integral_error, 10), -10)  # anti-windup

    # Step 3 — Update estimate

    correction      = Kp * error + Ki * integral_error

    estimated_angle = estimated_angle + correction

    return estimated_angle, integral_error

```
## Tuning Your Kp and Ki for GoKart

```

GoKart steering physical limits:

  Max steering rate  = 120°/sec  (fast hands)

  Typical cornering  = 30–60°/sec

  Settling time      = 0.2–0.5 seconds

Recommended starting values:

  Kp = 0.8   (tracks fast but not instant)

  Ki = 0.05  (slow integral — corrects bias gently)

  dt = 0.05  (20Hz AS5600 reading)

If path overshoots → reduce Kp

If path lags real steering → increase Kp

If steady drift remains → increase Ki

If oscillates → reduce Ki



import smbus2, math, time

# Constants

dt           = 0.05

# State

x = y = theta = 0.0

while True:

    # 1. Read raw steering angle from AS5600

    # 2. PI filter → smooth estimated angle

    error      = raw_angle - est_angle

    integ_err  = max(min(integ_err + error * dt, 10), -10)

    est_angle += Kp * error + Ki * integ_err

    # 3. Convert to road wheel angle

    delta_rad  = math.radians(est_angle / STEER_RATIO)

    # 4. Read speed from A3144 (your existing code)

    speed_ms   = read_hall_speed()

    # 5. Dead reckoning with PI-filtered angle

    dtheta     = (speed_ms / WHEELBASE) * math.tan(delta_rad) * dt

    theta     += dtheta

    x         += speed_ms * math.cos(theta) * dt

    y         += speed_ms * math.sin(theta) * dt

    time.sleep(dt)
